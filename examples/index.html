# Centered

# Usage

<template>
  <div class="wrapper">
    <button class="open" @click="openModal">
      Open
    </button>
  </div>
</template>
<script>
import BasicComponent from '../src/components/BasicComponent'

export default {
  methods: {
    openModal() {
      this.$FModal.show(
        { 
          component: BasicComponent, 
          width: 300, 
          maxHeight: '100%', 
          placement: 'center center' 
        }
      )
    }
  }
}
</script>

# Component Props

# Usage

      Open
    </button>
  </div>
</template>

import PropsComponent from '../src/components/PropsComponent'

export default {
  methods: {
    openModal() {
      this.$FModal.show(
        { 
          component: PropsComponent 
        }, 
        { 
          text: 'Hello Props', 
          color: '#7c8', 
          contentPadding: 30 
        }
      )
    }
  }
}
</script>
